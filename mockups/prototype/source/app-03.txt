ById('quick-estimate').value),due:'today',priority:'normal',recurrence:lower.includes('weekly')?'Weekly':'',grading:type==='chore'&&(lower.includes('clean')||lower.includes('wipe')),status:'open',nudge:true});save();overlayRoot.innerHTML='';route('today');showToast(`${title} added`);
    }
    else if(action==='process-gemini'){state.geminiText=document.getElementById('gemini-input')?.value||'';state.geminiStage='review';save();render();}
    else if(action==='cancel-gemini-review'){state.geminiStage='input';save();render();}
    else if(action==='confirm-gemini'){const changes=inferGemini(state.geminiText);changes.forEach(c=>{if(c.type==='list'){state.selectedList='groceries';const l=listById('groceries');if(!l.items.some(i=>i.name.toLowerCase()===c.title.toLowerCase()))l.items.push({id:'i'+Date.now()+Math.random(),name:c.title,quantity:'',checked:false,lastUsed:'New'});}else state.tasks.unshift({id:'t'+Date.now()+Math.random(),title:c.title,type:c.type,area:c.detail.includes('Car')?'Car':c.detail.includes('Bathroom')?'House':'Personal',room:c.detail.includes('Bathroom')?'Bathroom':'',estimate:10,due:c.detail.includes('Saturday')?'upcoming':'today',priority:'normal',recurrence:c.detail.includes('Monthly')?'Monthly':'',grading:c.type==='chore',status:'open',nudge:true});});state.geminiStage='input';state.geminiText='';save();route('today');showToast(`${changes.length} changes added`);}
    else if(action==='next-onboarding'){state.onboardingStep=Math.min(4,(state.onboardingStep||0)+1);save();render();}
    else if(action==='prev-onboarding'){state.onboardingStep=Math.max(0,(state.onboardingStep||0)-1);save();render();}
    else if(action==='finish-onboarding'){state.onboardingStep=0;save();route('today');showToast('Welcome to Nudge');}
    else if(action==='reset-demo')resetDemo();

    else if(action==='new-list')openNewList();
    else if(action==='save-area'){const name=document.getElementById('new-area-name')?.value.trim();if(!name){showToast('Enter an area name');return;}const rooms=(document.getElementById('new-area-rooms')?.value||'').split(',').map(x=>x.trim()).filter(Boolean);state.areas.push({id:'a'+Date.now(),name,icon:'□',rooms});save();overlayRoot.innerHTML='';route('areas');showToast(`${name} added`);}
    else if(action==='save-list'){const name=document.getElementById('new-list-name')?.value.trim();if(!name){showToast('Enter a list name');return;}const icon=document.getElementById('new-list-icon')?.value||'📝',id='l'+Date.now();state.lists.push({id,name,icon,lastUsed:'Never',items:[],catalog:[]});state.selectedList=id;save();overlayRoot.innerHTML='';route('list-detail');showToast(`${name} created`);}
    else if(action==='confirm-template'){const index=Number(overlayRoot.dataset.templateIndex||0),defs=[['Kitchen basics','Kitchen',['Take out trash','Wipe counters','Clean microwave','Mop floor','Clean refrigerator']],['Bathroom cleaning','Bathroom',['Clean sink','Clean mirror','Scrub shower','Replace towels']],['Bedroom reset','Main Bedroom',['Make bed','Put clothes away','Dust surfaces','Vacuum floor']],['Monthly car maintenance','',['Check tire pressure','Check washer fluid','Review oil level','Clean interior','Review registration','Check emergency kit']]];const d=defs[index]||defs[0];d[2].forEach((title,i)=>state.tasks.push({id:'tpl'+Date.now()+i,title,type:index===3?'task':'chore',area:index===3?'Car':'House',room:d[1],estimate:5+i*2,due:'upcoming',priority:'normal',recurrence:index===3?'Monthly':'Weekly',grading:index!==3,status:'open',nudge:i<3}));save();overlayRoot.innerHTML='';route(index===3?'tasks':'areas');showToast(`${d[0]} template added`);}
  });

  document.getElementById('status-time').textContent=new Date().toLocaleTimeString([], {hour:'numeric',minute:'2-digit'});
  render();
